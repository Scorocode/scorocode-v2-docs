    Sdk description

    Начало работы

    Убедитесь, пожалуйста, что у вас установлены:

    node.js - https://nodejs.org/en/download/

    tslib - https://www.npmjs.com/package/tslib

    npm - https://www.npmjs.com/get-npm

    Установка SDK

    npm i @scorocode/client-sdk

    Импорт SDK

    CommonJS

    const sc = require('@scorocode/client-sdk').default

    ES2015

    import sc from '@scorocode/client-sdk'

    Браузер

    <script src="https://accounts.scorocode.ru/downloads/client-sdk/1.0.0/scorocode.js">
</script>
<script>
scorocode.initApp({ ... })
</script>

    Менеджер приложений

    Обращение к менеджеру приложений

    const sc = require('@scorocode/client-sdk').default

    В браузерном окружении менеджер приложений будет доступен из глобальной переменной window.scorocode

    Инициализация приложения по умолчанию

    const app = sc.initApp({
  appId: '654...',
})

    Обращение к приложению по умолчанию

    Для обращения к приложению по умолчанию указывать идентификатор не нужно (если приложение одно)

    cosnt appRef = sc.app()

    Удаление приложения по умолчанию

    sc.removeApp().then(() => {
  console.log('Default application has been removed!')
})

    Создание приложения с идентификатором

    const anotherApp = sc.initApp({ appId: '213...' }, 'anotherApp')

    Обращение к приложению по идентификатору

    const anotherAppRef = sc.app('anotherApp')

    Удаление приложения по идентификатору

    sc.removeApp('anotherApp').then(() => {
  console.log('An "anotherApp" application has been removed!')
})

    Приложение

    Обращение к приложению

    const sc = require('@scorocode/client-sdk').default
const app = sc.initApp({ ... })

    Синхронное получение сервиса (без конфигурации сервиса)

    Если сервис не был создан, то он будет создан с конфигурацией по умолчанию и возвращен let pgService = sc.app().pg()

    Асинхронное получение сервиса

    Если сервис не был создан, он будет создан с переданной конфигурацией поверх конфигурации по умолчанию и возвращен

    let authService = await sc.app().auth({ ... })

    Если код приложения был подписан на события сервиса, нужно переподписаться в новой версии сервиса

    authService = await sc.app().auth({ ... })

    СОБЫТИЯ ПРИЛОЖЕНИЯ

    Событие срабатывает при каждом обращении к API приложения

    app.on('onRequest', (request) => {
})

    Событие срабатывает при успешном запросе к API приложения

    app.on('onRequestSuccess', (response, request) => {
})

    Событие срабатывает при ошибке во время выполнения запроса к API приложения

    app.on('onRequestError', (error, request) => {
})

    Сервис авторизации

    Инициализация приложения

    const auth = sc.app().auth()

    Получение текущей сессии

    const user = auth.currentSession

    Получение информации о пользователе

    const user = session.info

    Регистрация пользователя

    auth.signUp('user@mail.com', '123qwe').then((session) => {
  console.log('Registered and authenticated user', session.user)
})

    Аутентификация пользователя

    auth.signIn('user@mail.com', '123qwe').then((session) => {
  console.log('Authenticated user', session.user)
})

    Разлогинивание пользователя

    auth.signOut().then(() => {
  console.log('User logged out')
})

    Обновление сессии пользователя

    auth.refresh().then((session) => {
  console.log('The user has refreshed session', session)
})

    Описание токенов

    const token = session.token                 // токен авторизации
const refreshToken = session.refreshToken   // токен обновления сессии
const isExpired = session.isExpired         // сессия истекла

    Работа с файловой системой

    Создание указателя на папку

    const folder = fs.folder('path/to/folder')

    Получение информации о папке

    folder.sync().then((folder) => {
  console.log('File list', folder.files)  // список указателей на папки/файлы
})

    Создание папки

    folder.create().then((folder) => {
  console.log('Folder has been created')
})

    Переименование папки

    folder.rename('newPath/to/folder').then((folder) => {
  console.log('Path', folder.path)  // 'newPath/to/folder'
})

    Удаление папки

    folder.delete().then((folder) => {
  console.log('Folder has been deleted')
})

    Создание указателя на файл

    const file = fs.file('path/to/file')

    Получение информации о файле

    file.sync().then((file) => {
  console.log('File list', file.content)  // Blob буфер с контентом файла
})

    Загрузка текста

    file.upload('text content').then((file) => {
  console.log(file.content)  // 'text content'
})

    Загрузка файла

    file.upload(new Blob(['file content'])).then((file) => {
  console.log('File has been uploaded')
})

    Переименование файла

    file.rename('newPath/to/file').then((file) => {
  console.log('Path', file.path)  // 'newPath/to/file'
})

    Удаление файла

    file.delete().then((file) => {
  console.log('File has been deleted')
})

    Работа с базами данных

    Получение объекта-ссылки на коллекцию/таблицу

    let table = sc.app().pgManager().table('dbId', 'schemaName', 'tableName')
// или
table = sc.app().pgManager().database('dbId').schema('schemaName').table('tableName')

    Создание ссылки на запись таблицы

    const record = sc.app().pgManager().database('dbId').schema('schemaName').table('tableName').record({ id: 123 })

    Обращение к записям

    Создание объекта-ссылки на новую запись

    const record = sc.app().pg().record('dbId', 'schemaName', 'tableName')

    Создание объекта-ссылки на запись в таблице

    const record = sc.app().pgManager().database('dbId').schema('schemaName').table('tableName').record({ id: 123 })

    Получение объекта-ссылки на запись в таблице

    const ref = { id: 1234 }  // объект описывающий ключи по которым будет искаться запись
let record = sc.app().pg().record('dbId', 'schemaName', 'tableName', ref)
// либо
let record = sc.app().pg().database('dbId').schema('schemaName').table('tableName').record(ref)

    Загрузка данных записи

    record.sync().then((record) => {
  console.log(record.attributes)    // данные записи { fieldName => fieldValue }
})

    Сохранить изменения в записи

    record.set('price', 100)
record.save().then((record) => {
  console.log(record.attributes) // { price: 100, ... }
})

    Создать новую запись

    const newRecord = sc.app().pg().record('dbId', 'testSchema', 'usersTable')
newRecord.attributes = { foo: 'bar' }
newRecord.save().then((record) => {
  console.log(record.ref)   // table's primary keys object { pk1: 123, pk2: 'as3dfa...' }
  console.log(record.attributes) // { id: '123...', foo: 'bar' }
})

    Удаление записи

    record.delete().then((record) => {
  console.log('Record deleted')
  console.log(record.isDeleted) // true
})

    РАБОТА С ДАННЫМИ ЗАПИСИ

    Получение объекта с данными записи

    Данные не передаются по ссылке, создается копия данных

    const record = sc.app().pg().record('myDb', 'test', 'users')  // новая запись
record.attributes = { price: 10 } // установление значения
const attributes = record.attributes  // получение значения
console.log(attributes) // { price: 10 }
// любые изменения полученного объекта с данными не приведут к изменению записи
attributes.price = 20
console.log(record.attributes) // { price: 10 }

    Установление значения записи

    record.attributes = { name: 'Product', price: 100 }
// или
record.setAttributes({ name: 'Product', price: 100 })

    Установление значения поля записи

    record.set('price', 55)

    Получение значения поля записи

    console.log(record.get('price')) // 55

    Рекурсивное объединение данных (merge)

    record.attributes = {
  a: 'A',
  b: 'B',
  o: {
    foo: 'foo'
  },
}
record.merge({
  b: 'b',
  o: {
    foo: 'FOO',
    bar: 'bar',
  },
  c: 'c'
})

    Объединение данных (assign)

    record.attributes = {
  a: 'A',
  b: 'B',
  o: {
    foo: 'foo'
  },
}
record.assign({
  b: 'b',
  o: {
    bar: 'bar',
  },
  c: 'c'
})

    Запросы к БД

    Создание объекта-запроса к БД

    const query = sc.app().pg().query('dbId', 'schemaName', 'tableName')

    Работа с запросом

    query
  .from(10)                       // получить записи после 10ой
  .limit(15)                      // лимит в 15 записей
  .orderBy('name', 'ascend')      // сортировать по имени по возрастанию
  .orderBy('price', 'descend')    // сортировать по цене по убыванию
  .filterBy({                     // фильтровать опубликованные записи
    isPublished: true,
  })
  .find()                         // получить записи
  .then((items) => {
    // items - массив записей (Record)
    console.log('Records', items)
  })

    Сервис веб-сокетов

    Подписка на сообщения

    ws.onMessage('post.updated', (message) => {
  console.log(message)
})

    Подписка на одно сообщение

    ws.onMessageOnce('post.updated', (message) => {
})

    Отписать коллбек от события

    Коллбек (от англ. callback) – это функция, которую мы передаём куда-либо и ожидаем, что она будет вызвана при наступлении события.

    ws.removeMessageListener('hello', callbackFn)

    Подписать коллбек на событие

    ws.onMessage('hello', helloHandler)

    Обращение к коллбеку

    const helloHandler = (message) => {}

    Отправка сообщения (любой длины)

    const payload = { foo: bar } // массивы, объекты, простые типы
ws.sendMessage('test', payload) // если соединение разорвано, сообщение не будет отправлено

    Отправка сообщения с подтверждением

    const payload = { foo: bar } // массивы, объекты, простые типы
// если соединение разорвано, сообщение будет добавлено в стек и отправлено при восстановлении соединения
ws.sendMessageSafe('test', payload, () => {
  console.log('Message has been sent successfully')
})
